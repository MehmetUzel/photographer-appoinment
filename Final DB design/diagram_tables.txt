// Creating tables
Table user as U {
  id int [pk, increment] // auto-increment
  first_name varchar
  last_name varchar
  email varchar
  phone varchar
  password varchar
  created_at timestamp
}

Table address {
  id int [pk]
  user_id int [ref: > U.id]
  name varchar
  city varchar [ref: > city.id]
  district varchar [ref: > district.id]
  street varchar
  neighbourhood varchar
  building_num varchar
  flat_num varchar
  instructions varchar
 }

  id int [pk]

  name varchar
 }

 }
